import yaml
import re
import sys

def clean_rule_string(rule_str):
    """
    清洗规则字符串，去除首尾空白和可能的引号包裹，
    使其符合格式二的纯字符串格式。
    """
    rule_str = rule_str.strip()
    # 如果整个规则被单引号或双引号包裹，去除它们（因为写入YAML时库会自动处理引号）
    if (rule_str.startswith('"') and rule_str.endswith('"')) or \
       (rule_str.startswith("'") and rule_str.endswith("'")):
        # 简单去除首尾引号，不处理内部转义，因为yaml.dump会重新处理
        rule_str = rule_str[1:-1]
    return rule_str

def parse_rule_field(rule_value):
    """
    解析 'rule' 字段。
    格式一通常是长字符串，包含 || 逻辑。
    格式二倾向于将 || 分隔的条件拆分为列表项。
    """
    if not isinstance(rule_value, str):
        return [str(rule_value)]

    # 简单的启发式拆分：按 || 分割
    # 注意：这假设 || 不在引号内部作为内容出现。
    # 对于大多数指纹规则，这种简单的分割是有效的。
    parts = rule_value.split('||')
    
    processed_parts = []
    for part in parts:
        cleaned = clean_rule_string(part)
        if cleaned:
            processed_parts.append(cleaned)
            
    return processed_parts

def convert_format1_to_format2(input_file, output_file):
    try:
        # 1. 读取格式一
        with open(input_file, 'r', encoding='utf-8') as f:
            # 使用 safe_load 加载 YAML
            data_list = yaml.safe_load(f)

        if not isinstance(data_list, list):
            print("错误: 输入文件不是 YAML 列表格式。")
            return

        # 2. 转换为格式二的字典结构
        format2_data = {}

        for item in data_list:
            name = item.get('name')
            rule = item.get('rule')
            
            if not name:
                continue
                
            # 解析规则
            rules_list = parse_rule_field(rule)
            
            # 格式二的结构是: Name: [ "rule1", "rule2" ]
            format2_data[name] = rules_list

        # 3. 写入格式二
        # default_flow_style=False 强制使用块风格 (列表形式)，即每行一个规则
        with open(output_file, 'w', encoding='utf-8') as f:
            yaml.dump(format2_data, f, allow_unicode=True, sort_keys=False, default_flow_style=False)
            
        print(f"✅ 转换成功！")
        print(f"📂 输入文件: {input_file}")
        print(f"💾 输出文件: {output_file}")
        print(f"📝 共处理 {len(format2_data)} 条规则。")

    except FileNotFoundError:
        print(f"错误: 找不到文件 '{input_file}'")
    except yaml.YAMLError as e:
        print(f"YAML 解析错误: {e}")
    except Exception as e:
        print(f"发生未知错误: {e}")

if __name__ == "__main__":
    # 默认文件名配置
    input_filename = 'format1.yaml'
    output_filename = 'format2.yaml'

    # 允许通过命令行参数指定文件名
    if len(sys.argv) > 1:
        input_filename = sys.argv[1]
    if len(sys.argv) > 2:
        output_filename = sys.argv[2]

    convert_format1_to_format2(input_filename, output_filename)
